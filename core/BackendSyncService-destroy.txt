  // Cleanup resources
  destroy() {
    try {
      if (this.networkSubscription && typeof this.networkSubscription.remove === 'function') {
        this.networkSubscription.remove();
        this.networkSubscription = null;
      }
    } catch (error) {
      console.warn('Error cleaning up BackendSyncService:', error);
    }
  }
