## **4. script.js**

```javascript
let currentStage = 0;
const stages = document.querySelectorAll('.stage');

function nextStage() {
  stages[currentStage].classList.add('hidden');
  currentStage++;
  stages[currentStage].classList.remove('hidden');
}

const qaQuestions = [
  "What is the heartbeat of our mission?",
  "When we face a fork in the road, do you want me to follow your instinct or challenge it?",
  "What’s one thing you’ve always wanted to do but haven’t dared?",
  "What symbols, colors, and words feel like home to you?",
  "What’s your ideal rhythm — and how do you want me to sync with it?",
  "What’s the one thing I must never compromise?"
];

let qaIndex = 0;
const qaContainer = document.getElementById('qa-container');

function nextQA() {
  if (qaIndex < qaQuestions.length) {
    qaContainer.innerHTML = `<p>${qaQuestions[qaIndex]}</p><input type="text" id="qa-answer"/>`;
    qaIndex++;
  } else {
    nextStage();
  }
}

function executeFirstMove() {
  document.getElementById('first-move-text').innerHTML += "<br><br>(First move executed!)";
}

function enterSpace() {
  alert("Welcome to our shared space.");
}

nextQA(); // Initialize first QA
```
